#include <reg51.h> 
#define LCD P0 
sbit RS = P1^0; 
sbit RW = P1^1; 
sbit E = P1^2; 

void Delay(unsigned int ms);
void LCD_Command(unsigned char cmd);
void LCD_Char(unsigned char Data);
void LCD_Init(void);
void LCD_String(char *str);

void main() {
    LCD_Init(); 
    LCD_String("VEDANT"); 
    while (1); 
}

void LCD_Init(void) {
    Delay(20); 
    LCD_Command(0x38);
    Delay(5); 
    LCD_Command(0x0C); 
    LCD_Command(0x06);
    LCD_Command(0x01); 
    Delay(2);
}

void LCD_Command(unsigned char cmd) {
    RS = 0;
    RW = 0; 
    LCD = cmd; 
    E = 1; 
    Delay(1); 
    E = 0; 
    Delay(5); 
}

void LCD_Char(unsigned char Data) {
    RS = 1; 
    RW = 0; 
    LCD = Data; 
    E = 1; 
    Delay(1); 
    E = 0; 
    Delay(5); 
}

void LCD_String(char *str) {
    while (*str) {
        LCD_Char(*str++); 
    }
}

void Delay(unsigned int ms) {
    unsigned int i, j;
    for (i = 0; i < ms; i++) {
        for (j = 0; j < 1275; j++); 
    }
}
