//blinking
#include <reg51.h>

void Delay(unsigned int time);

void main() {
    P1 = 0x00; 
    while (1) {
        P1 = 0xFF; // Turn ON 
        Delay(250);
        P1 = 0x00; // Turn OFF 
        Delay(250);
    }
}

void Delay(unsigned int time) {
    unsigned int i, j;
    for (i = 0; i < time; i++)
        for (j = 0; j < 165; j++); 
}
//toggling
#include <reg51.h>

void Delay(unsigned int time);

void main() {
    while(1) {
        P1 = 0xAA;     
        Delay(250);
        P1 = ~P1;      
        Delay(250);
    }
}

void Delay(unsigned int time) {
    unsigned int i, j;
    for(i = 0; i < time; i++)
        for(j = 0; j < 165; j++); 
}
